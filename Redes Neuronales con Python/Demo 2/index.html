# 1. ARQUITECTURA DE LA RED (Keras/TensorFlow)
# Diseñamos la estructura capa por capa

model = Sequential([
    # Capa de Entrada: Recibe datos brutos
    Dense(units=8, activation='relu', input_shape=(3,)),
    
    # Capas Ocultas: Procesamiento profundo
    Dense(units=8, activation='relu'),
    Dense(units=8, activation='relu'),
    
    # Capa de Salida: Clasificación final (2 clases)
    Dense(units=2, activation='softmax')
])

# 3. OPTIMIZACIÓN DE HIPERPARÁMETROS
# Probamos múltiples combinaciones automáticamente

# Definir qué opciones probar
param_grid = {
    'learning_rate': [0.01, 0.001],
    'batch_size': [16, 32, 64]
}

# Ejecutar búsqueda exhaustiva (Grid Search)
search = GridSearchCV(estimator=model, param_grid=param_grid)
result = search.fit(x_train, y_train)

print(f"Mejor configuración: {result.best_params_}")
